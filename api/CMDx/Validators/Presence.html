def call(value, options = EMPTY_HASH)
  present =
    if value.is_a?(String)
      /\S/.match?(value)
    elsif value.respond_to?(:empty?)
      !value.empty?
    else
      !value.nil?
    end

  return if present

  Failure.new(options[:message] || I18nProxy.t("cmdx.validators.presence"))
end
