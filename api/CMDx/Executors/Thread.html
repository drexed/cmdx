Default executor. Uses a fixed-size Thread pool drained via a Queue; sentinel nils terminate workers. Workers inherit the parent's chain via fiber-local storage.

def call(jobs:, concurrency:, on_job:)
  queue = Queue.new
  jobs.each { |job| queue << job }
  concurrency.times { queue << nil }

  workers = Array.new(concurrency) do
    ::Thread.new do
      while (job = queue.pop)
        on_job.call(job)
      end
    end
  end

  workers.each(&:join)
end
